<label for="gram">섭취량(g):</label>

<input type="number" id="gram" placeholder="예: 150"><br>



<button onclick="calculate()">칼로리 계산</button>



<div class="result" id="result"></div>



<script>

function calculate() {

const food = document.getElementById("food");

const gram = parseFloat(document.getElementById("gram").value);

const kcalPer100g = parseFloat(food.options[food.selectedIndex].dataset.cal);



if (isNaN(gram) || kcalPer100g === 0) {

document.getElementById("result").innerHTML = "음식과 섭취량을 정확히 입력하세요.";

return;

}



const total = (gram / 100) * kcalPer100g;

document.getElementById("result").innerHTML =

`총 섭취 칼로리는 <strong>${total.toFixed(1)} kcal</strong>입니다.`;

}

</script>
